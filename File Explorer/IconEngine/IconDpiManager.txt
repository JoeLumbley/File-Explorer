' Central place for DPI-aware icon sizing and cache versioning.
' This keeps DPI logic out of the UI and out of ShellInterop.
Public Class IconDpiManager

    ' Current DPI of the window/monitor (96 = 100%).
    Public Shared Property CurrentDpi As Integer = 96

    ' Incremented whenever DPI changes so caches can invalidate.
    Public Shared Property CacheVersion As Integer = 0

    ' Converts a logical icon size (16, 32, etc.) to a DPI-scaled pixel size.
    Public Shared Function ToPixelSize(logicalSize As Integer) As Integer
        If logicalSize <= 0 Then logicalSize = 16

        Dim scale As Double = CurrentDpi / 96.0
        Dim raw As Double = logicalSize * scale

        ' Match Explorer behavior: round to nearest pixel.
        Return CInt(Math.Round(raw, MidpointRounding.AwayFromZero))
    End Function

    ' Called when the window’s DPI changes.
    ' Form1 (or a controller) should call this from its DPI change handler.
    Public Shared Sub OnDpiChanged(newDpi As Integer)
        ' Ignore invalid or unchanged DPI.
        If newDpi <= 0 OrElse newDpi = CurrentDpi Then Exit Sub

        CurrentDpi = newDpi
        CacheVersion += 1

        ' Clear icon cache so icons can be regenerated at new DPI.
        IconCache.Clear()
    End Sub

End Class
