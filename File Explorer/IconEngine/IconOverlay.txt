Imports System.Drawing
Imports System.Drawing.Imaging
Imports System.Drawing.Drawing2D

' Applies overlay icons (shortcut arrow, shared, etc.) on top of base icons.
' This is purely visual; it doesn’t affect caching keys.
Public Class IconOverlay

    ' Returns a new icon with the specified overlay applied.
    ' If no overlay is requested, the base icon is returned as-is.
    Public Shared Function ApplyOverlay(baseIcon As Icon, overlay As IconOverlayKind) As Icon
        If baseIcon Is Nothing OrElse overlay = IconOverlayKind.None Then
            Return baseIcon
        End If

        Dim size = baseIcon.Size
        Dim bmp As New Bitmap(size.Width, size.Height, PixelFormat.Format32bppArgb)

        Using g = Graphics.FromImage(bmp)
            ' High-quality rendering for overlays.
            g.InterpolationMode = InterpolationMode.HighQualityBicubic
            g.SmoothingMode = SmoothingMode.HighQuality
            g.PixelOffsetMode = PixelOffsetMode.HighQuality

            ' Draw the base icon.
            g.DrawIcon(baseIcon, New Rectangle(Point.Empty, size))

            ' Choose overlay icon.
            Dim overlayIcon As Icon = Nothing
            Select Case overlay
                Case IconOverlayKind.Shortcut
                    overlayIcon = IconLibrary.ShortcutOverlay
                Case IconOverlayKind.SharedFolder
                    overlayIcon = IconLibrary.SharedOverlay
            End Select

            ' Draw overlay in bottom-right corner.
            If overlayIcon IsNot Nothing Then
                Dim overlaySize As New Size(size.Width \ 2, size.Height \ 2)
                Dim overlayRect As New Rectangle(size.Width - overlaySize.Width,
                                                 size.Height - overlaySize.Height,
                                                 overlaySize.Width,
                                                 overlaySize.Height)
                g.DrawIcon(overlayIcon, overlayRect)
            End If
        End Using

        ' Create an Icon from the bitmap handle.
        ' Note: lifetime of this icon is managed by the caller / cache.
        Dim hIcon = bmp.GetHicon()
        Return Icon.FromHandle(hIcon)
    End Function

End Class
